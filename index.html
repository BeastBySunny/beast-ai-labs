const express=require("express") const cors=require("cors") const OpenAI=require("openai") const nodemailer=require("nodemailer") const twilio=require("twilio") const app=express() app.use(cors()) app.use(express.json()) const openai=new OpenAI({ apiKey:"YOUR_OPENAI_KEY" }) app.post("/api/ask",async(req,res)=>{ const question=req.body.question const response=await openai.chat.completions.create({ model:"gpt-4o-mini", messages:[ {role:"system",content:"You are an AI assistant for Beast AI Labs."}, {role:"user",content:question} ] }) res.json({ answer:response.choices[0].message.content }) }) app.post("/api/contact",async(req,res)=>{ const {name,email,message}=req.body /* EMAIL */ const transporter=nodemailer.createTransport({ service:"gmail", auth:{ user:"YOUR_EMAIL", pass:"EMAIL_APP_PASSWORD" } }) await transporter.sendMail({ from:email, to:"YOUR_EMAIL", subject:"New Inquiry", text:`Name:${name} Email:${email} Message:${message}` }) /* WHATSAPP */ const client=twilio( "TWILIO_ACCOUNT_SID", "TWILIO_AUTH_TOKEN" ) await client.messages.create({ from:"whatsapp:+14155238886", to:"whatsapp:+919971260681", body:`New website inquiry Name:${name} Email:${email} Message: ${message}` }) res.json({status:"sent"}) }) app.listen(3000,()=>{ console.log("Server running") })
